### 기억해!!!!

#### combination은 30초 안에 만들어야함

```js
/**
 *
 * @param arr 조합 구할 배열
 * @param r 선택할 개수
 * @param index 현재 배열에 접근할 index
 * @param cur 현재 만들고 있는 조합 배열
 * @returns
 */
function combination(arr, r, index, cur) {
  if (cur.length === r) {
    console.log(cur);
    return;
  }

  for (let i = index; i < arr.length; i++) {
    cur.push(arr[i]);
    combination(arr, r, i + 1, cur); // 중복 조합은 i + 1 대신 i
    cur.pop();
  }
}
```
