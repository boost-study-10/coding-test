### 🧠 대략적인 코드 설명
```js
const input = require("fs")
  .readFileSync("/dev/stdin")
  .toString()
  .trim()
  .split("\n");

const pattern = input[1].split('*').map(it => it.trim());

function sol() {
    const pre = pattern[0];
    const post = pattern[1];

    for (let i = 2; i < input.length; i++) {
        const target = input[i];
        if (target.length >= pre.length + post.length && target.startsWith(pre) && target.endsWith(post)) console.log("DA");
        else console.log("NE");
    }
}

sol();
```
