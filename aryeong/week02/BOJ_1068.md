### 🧠 대략적인 코드 설명
```js
const input = require("fs")
    .readFileSync("/dev/stdin")
    .toString()
    .trim()
    .split("\n");

const n = Number(input[0]);
const parents = input[1].trim().split(' ').map(Number);
const delNum = Number(input[2]);

const tree = Array.from({ length: n }, () => []);

for (let i = 0; i < n; i++) {
    const parent = parents[i];
    if (parent !== -1) {
        tree[parent].push(i);
    }
}

function dfs(node) {
    let leaf = 0;
    let childCount = 0;

    for (const next of tree[node]) {
        if (next === delNum) continue;
        leaf += dfs(next);
        childCount++;
    }

    if (childCount === 0) return 1;
    return leaf;
}

let root = parents.indexOf(-1);

if (root === delNum) {
    console.log(0);
} else {
    console.log(dfs(root));
}
```
