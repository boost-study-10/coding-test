### 📖 풀이한 문제
- 백준 16234 - 인구 이동
- 문제 링크: https://www.acmicpc.net/problem/16234
- 난이도: 골드4
- 걸린 시간: 35분

### 🧩 풀이 설계 / 막힌 부분
- 처음 접근 아이디어: 1. 모든 나라를 순회하면서 방문하지 않은 나라를 시작점으로 DFS 2. 인구 차이가 L이상 R이하인 나라끼리 연합 생성 3. 연합 국가 총 인구 합 구하기 4. 연합 인구 평균 구해서 국가 인구 갱신 5. 인구 이동이 발생했으면 다음 날 한 번 더

### ⭐️ 문제에서 주로 사용한 알고리즘
- dfs

### 🧠 대략적인 코드 설명
```js
const input = require("fs")
    .readFileSync("/dev/stdin")
    .toString()
    .trim()
    .split("\n");

const [n, l, r] = input[0].split(' ').map(Number);
const arr = input.slice(1).map((line) => line.split(' ').map(Number));

const dy = [0, 1, 0, -1];
const dx = [1, 0, -1, 0];

let visited = Array.from({ length: n }, () => Array(n).fill(0));
let sum = 0;

function dfs(y, x, v) {
    for (let i = 0; i < 4; i++) {
        const ny = y + dy[i];
        const nx = x + dx[i];
        if (ny < 0 || nx < 0 || ny >= n || nx >= n || visited[ny][nx]) continue;
        const diff = Math.abs(arr[y][x] - arr[ny][nx]);
        if (diff >= l && diff <= r) {
            visited[ny][nx] = 1;
            v.push([ny, nx]);
            sum += arr[ny][nx];
            dfs(ny, nx, v);
        }
    }
}

let cnt = 0;

while(true) {
    let flag = 0;
    visited = Array.from({ length: n }, () => Array(n).fill(0));
    let union = [];
    for (let i = 0; i < n; i++) {
        for (let j = 0; j < n; j++) {
            if (visited[i][j]) continue;
            union = [];
            visited[i][j] = 1;
            union.push([i, j]);
            sum = arr[i][j];
            dfs(i, j, union);
            if(union.length === 1) continue;
            for (const [i, j] of union) {
                arr[i][j] = Math.floor(sum / union.length);
                flag = 1;
            }
        }
    }

    if(!flag) break;
    cnt++;
}

console.log(cnt);
```
