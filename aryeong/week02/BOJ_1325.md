### ⭐️ 문제에서 주로 사용한 알고리즘
- bfs

### 🧠 대략적인 코드 설명
```js
const input = require("fs")
    .readFileSync("example.txt")
    .toString()
    .trim()
    .split("\n");

const [n, m] = input[0].split(' ').map(Number);
const adj = Array.from({ length: n + 1 }, () => []);

for (let i = 1; i <= m; i++) {
    const [a, b] = input[i].split(' ');
    adj[Number(b)].push(Number(a));
}

function bfs(start) {
    const visited = new Array(n + 1).fill(0);
    visited[start] = 1;
    
    const queue = [start];
    let idx = 0;
    let cnt = 0;

    while (idx < queue.length) {
        const node = queue[idx++];
        for (const next of adj[node]) {
            if (!visited[next]) {
                visited[next] = 1;
                queue.push(next);
                cnt++;
            }
        }
    }

    return cnt;
}

let max = 0;
let nums = [];
for (let i = 1; i <= n; i++) {
    const ret = bfs(i);
    if (ret > max) {
        max = ret;
        nums = [i];
    } else if (ret === max) {
        nums.push(i);
    }
}

console.log(nums.join(' '));
```
