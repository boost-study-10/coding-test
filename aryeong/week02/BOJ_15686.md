### 📖 풀이한 문제
- 백준 15686 - 치킨 배달
- 문제 링크: https://www.acmicpc.net/problem/15686
- 난이도: 골드5
- 걸린 시간: 40분

### 🧩 풀이 설계 / 막힌 부분
- 처음 접근 아이디어: 1. 일단 치킨집의 좌표를 구한다. 2. 치킨집 중 M개를 백트래킹으로 조합한다. 3. 조합 하나당 도시의 치킨 거리를 구한다. 도시 치킨 거리 = 각 집 최소거리 합. 4. min 치킨 거리인 경우를 구한다.
- 고려한 시간복잡도: C(치킨집, m) * 집 개수 * m
- 막힌 부분 / 해결 방법:

### ⭐️ 문제에서 주로 사용한 알고리즘
- 백트래킹

### 🧠 대략적인 코드 설명
```js
const input = require("fs")
    .readFileSync("/dev/stdin")
    .toString()
    .trim()
    .split("\n");

const [n, m] = input[0].split(' ').map(Number);
const arr = input.slice(1).map((line) => line.split(' ').map(Number));
const chickens = [];
const houses = [];

for (let i = 0; i < n; i++) {
    for (let j = 0; j < n; j++) {
        if (arr[i][j] === 2) chickens.push([i, j]);
        else if (arr[i][j] === 1) houses.push([i, j]);
    }
}

function getDistance(selectedChickens) {
    let total = 0;

    for (let i = 0; i < houses.length; i++) {
        let temp = Infinity;
        for (let j = 0; j < selectedChickens.length; j++) {
            const distance = Math.abs(houses[i][0] - selectedChickens[j][0]) + Math.abs(houses[i][1] - selectedChickens[j][1]);
            temp = Math.min(temp, distance);
        }
        total += temp;
    }
    return total;
}

let ans = Infinity;

function combi(start, arr) {
    if (arr.length === m) {
        const distance = getDistance(arr);
        ans = Math.min(ans, distance);
    }

    for (let i = start; i < chickens.length; i++) {
        arr.push(chickens[i]);
        combi(i + 1, arr);
        arr.pop();
    }
}

combi(0, []);

console.log(ans);
```
