### 📖 풀이한 문제
- 백준 2589 - 보물섬
- 문제 링크: https://www.acmicpc.net/problem/2589
- 난이도: 골드5
- 걸린 시간: 20분

### 🧩 풀이 설계 / 막힌 부분
- 처음 접근 아이디어: 육지에서 가장 먼 육지까지의 거리를 구하면 된다. 모든 육지에서 BFS -> 최장거리 구함
- 고려한 시간복잡도: 50 * 50
- 막힌 부분 / 해결 방법:

### ⭐️ 문제에서 주로 사용한 알고리즘
- bfs

### 🧠 대략적인 코드 설명
```js
const input = require("fs")
    .readFileSync("example.txt")
    .toString()
    .trim()
    .split("\n");

const [n, m] = input[0].split(' ').map(Number);
const arr = input.slice(1).map((line) => line.trim().split(''));

const dy = [0, 1, 0, -1];
const dx = [1, 0, -1, 0];

function bfs(a, b) {
    const visited = Array.from({ length: n }, () => Array(m).fill(0));
    const queue = [[a, b]];
    visited[a][b] = 1;
    let idx = 0;
    let max = 0;

    while (idx < queue.length) {
        const [y, x] = queue[idx++];
        for (let i = 0; i < 4; i++) {
            const ny = y + dy[i];
            const nx = x + dx[i];
            if (ny < 0 || nx < 0 || ny >= n || nx >= m || visited[ny][nx]) continue;
            if (arr[ny][nx] === 'W') continue;
            queue.push([ny, nx]);
            visited[ny][nx] = visited[y][x] + 1;
            max = Math.max(max, visited[ny][nx]);
        }
    }

    return max - 1;
}

let ans = 0;
for (let i = 0; i < n; i++) {
    for (let j = 0; j < m; j++) {
        if (arr[i][j] === 'L') {
            ans = Math.max(ans, bfs(i, j));
        }
    }
}

console.log(ans);
```
