- dfs, 백트래킹

### 🧠 대략적인 코드 설명
```js
const input = require("fs")
  .readFileSync("/dev/stdin")
  .toString()
  .trim()
  .split("\n");

const [r, c] = input[0].split(' ').map(Number);
const adj = input.slice(1).map((v) => v.trim().split(''));

const dy = [1, 0, -1, 0];
const dx = [0, 1, 0, -1];

const st = new Set();
let max = 0;
function dfs(y, x, cnt) {
    max = Math.max(max, cnt);

    for (let i = 0; i < 4; i++) {
        const ny = y + dy[i];
        const nx = x + dx[i];
        if (ny < 0 || ny >= r || nx < 0 || nx >= c) continue;
        if (!st.has(adj[ny][nx])) {
            st.add(adj[ny][nx]);
            dfs(ny, nx, cnt + 1);
            st.delete(adj[ny][nx]);
        }
    }
}

st.add(adj[0][0]);
dfs(0, 0, 1);

console.log(max);
```
