### 🧠 대략적인 코드 설명
```js
const input = require("fs")
    .readFileSync("/dev/stdin")
    .toString()
    .trim()
    .split("\n");

const [r, c] = input[0].split(' ').map(Number);
const adj = input.slice(1).map((line) => line.trim().split(''));

const dy = [0, 1, 0, -1];
const dx = [1, 0, -1, 0];

const fireDistance = Array.from({ length: r }, () => Array(c).fill(Infinity));
function fire() {
    const queue = [];
    for (let i = 0; i < r; i++) {
        for (let j = 0; j < c; j++) {
            if (adj[i][j] === 'F') {
                queue.push([i, j]);
                fireDistance[i][j] = 0;
            }
        }
    }

    let idx = 0;

    while (idx < queue.length) {
        const [y, x] = queue[idx++];
        for (let i = 0; i < 4; i++) {
            const ny = y + dy[i];
            const nx = x + dx[i];
            if (ny < 0 || nx < 0 || ny >= r || nx >= c) continue;
            if (adj[ny][nx] === '#' || fireDistance[ny][nx] !== Infinity) continue;
            queue.push([ny, nx]);
            fireDistance[ny][nx] = fireDistance[y][x] + 1;
        }
    }
}

function bfs(a, b) {
    const visited = Array.from({ length: r }, () => Array(c).fill(-1));
    visited[a][b] = 0;
    const queue = [[a, b]];
    let idx = 0;

    while (idx < queue.length) {
        const [y, x] = queue[idx++];

        if (y === 0 || x === 0 || y === r - 1 || x === c - 1) {
            console.log(visited[y][x] + 1);
            return;
        }

        for (let i = 0; i < 4; i++) {
            const ny = y + dy[i];
            const nx = x + dx[i];
            if (ny < 0 || nx < 0 || ny >= r || nx >= c) continue;
            if (adj[ny][nx] === '#' || visited[ny][nx] !== -1) continue;
            if (fireDistance[ny][nx] <= visited[y][x] + 1) continue;

            visited[ny][nx] = visited[y][x] + 1;
            queue.push([ny, nx]);
        }
    }

    console.log('IMPOSSIBLE');
}

let start;
for (let i = 0; i < r; i++) {
    for (let j = 0; j < c; j++) {
        if (adj[i][j] === 'J') start = [i, j];
    }
}

fire();
bfs(start[0], start[1]);
```
