- 고려한 시간복잡도:
- 막힌 부분 / 해결 방법: 지금 루프를 3번 돌고 있는데 한번에 처리해도 좋을듯.. set말고 includes 사용도 가능

### 🧠 대략적인 코드 설명
```js
const input = require("fs")
    .readFileSync("/dev/stdin")
    .toString()
    .trim()
    .split("\n");

const passwords = input.map((v) => v.trim());

function hasVowel(word) {
    const st = new Set(word.split(''));
    if (st.has('a') || st.has('e') || st.has('i') || st.has('o') || st.has('u')) return true;
    return false;
}

function IsNotContinueThreetimes(word) {
    let cnt = 1;
    const st = new Set(['a', 'e', 'i', 'o', 'u']);
    let flag = st.has(word[0]);

    for (let i = 1; i < word.length; i++) {
        if (flag === st.has(word[i]) && cnt === 2) return false;
        else if(flag === st.has(word[i]) && cnt < 2) cnt++;
        else {
            flag = st.has(word[i]);
            cnt = 1;
        }
    }

    return true;
}

function IsNotContinueTwice(word) {
    for (let i = 0; i < word.length - 1; i++) {
        if (word[i] === word[i + 1] && !(word[i] === 'e' && word[i + 1] === 'e') && !(word[i] === 'o' && word[i + 1] === 'o')) return false;
    }

    return true;
}

const ret = [];
for (let i = 0; i < passwords.length - 1; i++) {
    const pw = passwords[i];
    const case1 = hasVowel(pw);
    if (!case1) {
        ret.push(false);
        continue;
    }
    const case2 = IsNotContinueThreetimes(pw);
    if (!case2) {
        ret.push(false);
        continue;
    }
    const case3 = IsNotContinueTwice(pw);
    if (!case3) {
        ret.push(false);
        continue;
    }

    ret.push(true);
}

for (let i = 0; i < ret.length; i++) {
    console.log(`<${passwords[i]}> is ${ret[i] ? '' : 'not '}acceptable.`);
}
```
