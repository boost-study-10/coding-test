### 🧠 대략적인 코드 설명
```js
const input = require("fs")
    .readFileSync("/dev/stdin")
    .toString()
    .trim()
    .split("\n");

let n = Number(input);
let num = 665;

while (n > 0) {
  num++;

  if (num.toString().includes("666")) {
    n--;
  }
}

console.log(num);
```
