### 📖 풀이한 문제
- 백준 14502 - 연구소
- 문제 링크: https://www.acmicpc.net/problem/14502
- 난이도: 골드4
- 걸린 시간: 45분

### 🧩 풀이 설계 / 막힌 부분
- 처음 접근 아이디어: 무작위로 3개 벽을 세운다. 벽 세운 lab에서 바이러스 확산 bfs로 한다. 해당 케이스의 안전 영역의 넓이(0의 개수)를 구한다. max로 매 케이스를 비교한다.

  - 벽은 어떻게 세우나..? => 백트래킹으로 or 0인 좌표 중 3개를 미리 조합한다.
- 고려한 시간복잡도: O(조합 개수 * NM)
- 막힌 부분 / 해결 방법: 바이러스 확산 후 안전영역 넓이는 전체에서 0의 개수 count만 하면 되는 거였는데 해당 케이스의 안전 영역을 bfs로 구하려고 함.. 최종 안전영역은 하나로 이어져있어서 그럴 필요가 없었음. 

### ⭐️ 문제에서 주로 사용한 알고리즘
- 백트래킹, bfs

### 🧠 대략적인 코드 설명
```js
const input = require("fs")
    .readFileSync("/dev/stdin")
    .toString()
    .trim()
    .split("\n");

const [n, m] = input[0].split(' ').map(Number);
const lab = input.slice(1).map((v) => v.trim().split(' ').map(Number));

const dy = [0, 1, 0, -1];
const dx = [1, 0, -1, 0];

let ans = 0;

function buildWall(count) {
    if(count === 3) {
        ans = Math.max(ans, solution());
        return;
    }

    for(let i = 0;i < n;i++) {
        for(let j = 0;j < m;j++) {
            if(lab[i][j] === 0) {
                lab[i][j] = 1;
                buildWall(count + 1);
                lab[i][j] = 0;
            }
        }
    }
}

function spreadVirus(copy) {
    const queue = [];
    let head = 0;

    for(let i = 0;i < n;i++) {
        for(let j = 0;j < m;j++) {
            if(copy[i][j] === 2) queue.push([i, j]);
        }
    }

    while (head < queue.length) {
        const [y, x] = queue[head++];

        for (let i = 0; i < 4; i++) {
            const ny = y + dy[i];
            const nx = x + dx[i];
            if (ny < 0 || nx < 0 || ny >= n || nx >= m) continue;
            if(copy[ny][nx] === 0) {
                copy[ny][nx] = 2;
                queue.push([ny, nx]);
            }
        }
    }
}

function countSafe(copy) {
    let cnt = 0;
    for(let i = 0;i < n;i++) {
        for(let j = 0;j < m;j++) {
            if(copy[i][j] === 0) cnt++;
        }
    }
    return cnt;
}

function solution() {
    const copy = lab.map((row) => [...row]);
    spreadVirus(copy);
    return countSafe(copy);
}

buildWall(0);
console.log(ans);
```
