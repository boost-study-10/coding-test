### 📖 풀이한 문제
- 백준 2583 - 영역 구하기
- 문제 링크: https://www.acmicpc.net/problem/2583
- 난이도: 실버1
- 걸린 시간: 30분

### 🧩 풀이 설계 / 막힌 부분
- 처음 접근 아이디어: 1. 모눈종이 2차원 배열에 직사각형 위치를 표시한다. 2. dfs로 돌면서 색칠되지 않은 부분이 연결된 지점을 찾는다. 이때 넓이도 같이 count.
- 고려한 시간복잡도: O(MN)
- 막힌 부분 / 해결 방법: 재귀로 area 누적

### ⭐️ 문제에서 주로 사용한 알고리즘
- dfs

### 🧠 대략적인 코드 설명
```js
const input = require("fs")
    .readFileSync("example.txt")
    .toString()
    .trim()
    .split("\n");

const [m, n, k] = input[0].split(' ').map(Number);
const sq = Array.from(Array(m), () => Array(n).fill(0));
const visited = Array.from(Array(m), () => Array(n).fill(0));

for (let i = 1; i < k + 1; i++) {
    const [x1, y1, x2, y2] = input[i].split(' ').map(Number);
    for (let x = x1; x < x2; x++) {
        for (let y = y1; y < y2; y++) {
            sq[y][x] = 1;
        }
    }
}

const nx = [1, 0, -1, 0];
const ny = [0, 1, 0, -1];
const areas = [];

function dfs(y, x) {
    visited[y][x] = 1;
    let area = 1;
    
    for (let i = 0; i < 4; i++) {
        const dy = y + ny[i];
        const dx = x + nx[i];
        if (dy < 0 || dx < 0 || dy >= m || dx >= n || sq[dy][dx]) continue;
        if (visited[dy][dx]) continue;
        area += dfs(dy, dx);
    }

    return area;
}

let ans = 0;
for(let i = 0;i < m;i++) {
    for(let j = 0;j < n;j++) {
        if(!sq[i][j] && !visited[i][j]) {
            const area = dfs(i, j);
            areas.push(area);
            ans++;
        }
    }
}

console.log(ans);
console.log(areas.sort((a, b) => a - b).join(' '));
```
